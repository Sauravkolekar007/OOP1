ORG 0000H

MOV DPTR, #PATTERN

MAIN:
    MOV R0, #08H

DISPLAY:
    MOV A, R0
    MOV P1, A
    MOVC A, @A+DPTR
    MOV P2, A
    ACALL DELAY
    MOV P2, #00H
    DJNZ R0, DISPLAY
    SJMP MAIN

DELAY:
    MOV R2, #50
D1: MOV R1, #255
D2: DJNZ R1, D2
    DJNZ R2, D1
    RET

PATTERN:
    DB 81H, 42H, 24H, 18H, 18H, 24H, 42H, 81H

END
