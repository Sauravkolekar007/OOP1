ORG 0H

START:
    MOV P1, #0FFH
    MOV P2, #00H
    
CHECK_SENSORS:
    MOV A, P1
    ANL A, #01H
    JZ LOW_LEVEL

    MOV A, P1
    ANL A, #02H
    JZ HIGH_LEVEL

    MOV A, P1
    ANL A, #04H
    JZ OVERFLOW_LEVEL

    SJMP START

LOW_LEVEL:
    SETB P2.0
    SJMP CHECK_SENSORS

HIGH_LEVEL:
    CLR P2.0
    SJMP CHECK_SENSORS

OVERFLOW_LEVEL:
    CLR P2.0
    SJMP CHECK_SENSORS

END
