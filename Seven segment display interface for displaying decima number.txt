ORG 0000H

MOV DPTR, #SEGMENT_CODES

MAIN:
    MOV R0, #0AH

LOOP:
    MOV A, R0
    MOVC A, @A+DPTR
    MOV P1, A
    ACALL DELAY
    INC R0
    CJNE R0, #0AH, LOOP
    MOV R0, #00H
    SJMP LOOP

DELAY:
    MOV R2, #100
D1: MOV R1, #255
D2: DJNZ R1, D2
    DJNZ R2, D1
    RET

SEGMENT_CODES:
    DB 3FH, 06H, 5BH, 4FH, 66H, 6DH, 7DH, 07H, 7FH, 6FH

END
